#include<iostream>
#include<limits.h>
using namespace std;

int a[] = {3,2,1, -1, -2, -3, 9};


void swap(int *m, int *n){
	int t;
	t = *m;
	*m=*n;
	*n=t;	
}

void merge(int p, int q, int r){
	int i, j, k;
    int n1 = q - p + 2;
    int n2 = r - q + 1;
 	int L[n1], R[n2];
	
	for(i=p, j=0;i<=q;i++){
		L[j++]=a[i];
	}
	L[j]=INT_MAX;
	
	for(i=q+1, j=0;i<=r;i++){
		R[j++]=a[i];
	}
	R[j]=INT_MAX;
	
	i=0;j=0;
	for(k=p;k<=r;k++){
		if(L[i]<=R[j]){
			a[k]=L[i++];
		}else{
			a[k]=R[j++];
		}
	}
} 

void ms(int p, int r){
	if(p<r){
		int q=(p+r)/2;
		ms(p,q);
		ms(q+1,r);		
		merge(p,q,r);
	}
}

int main(){
    ms(0,6);
    for(int i=0;i<7;i++)
    	cout<<a[i]<<"\t";
    return 0;
}
