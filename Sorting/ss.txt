#include<iostream>
#include<limits.h>
using namespace std;

int a[] = {3,2,1, -1, -2, -3, 0};


void swap(int *m, int *n){
	int t;
	t = *m;
	*m=*n;
	*n=t;	
}

void ss(){
	int j,l=sizeof(a)/sizeof(a[0]),min_indx,i;
	for(i=0;i<l-1;i++){
		
		min_indx = i;
		
		for(j=i+1;j<l;j++){
			if(a[min_indx]>a[j]){
				min_indx=j;
			}
		}
		
		swap(&a[i],&a[min_indx]);
	}
}

int main(){
    ss();
    for(int i=0;i<7;i++)
    	cout<<a[i]<<"\t";
    return 0;
}
