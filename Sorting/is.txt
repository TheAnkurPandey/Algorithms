#include<iostream>
#include<limits.h>
using namespace std;

int a[] = {3,2,1, -1, -2, -3, 0};

void is(){
	int j,l=sizeof(a)/sizeof(a[0]),key,i;
	for(j=1;j<l;j++){
		
		key = a[j];
		i = j-1;
		
		while(i>=0 && a[i]>key){
			a[i+1]=a[i];
			i--;
		}
		a[i+1]=key;
	}
}

int main(){
    is();
    for(int i=0;i<7;i++)
    	cout<<a[i]<<"\t";
    return 0;
}
